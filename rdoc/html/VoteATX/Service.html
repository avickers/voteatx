def search(params)
  lat = nil
  lng = nil
  query_opts = {}

  params.each do |k, v|
    k = k.to_sym
    case k
    when :latitude
      lat = v.to_f
    when :longitude
      lng = v.to_f
    when :time, :max_distance, :max_locations
      query_opts[k] = v
    end
  end

  result = @@app.search(lat, lng, query_opts)

  content_type :json
  jsonp result.map{|e| e.to_h}
end
